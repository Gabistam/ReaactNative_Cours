import { initializeApp }        from 'firebase/app';
import { getAuth }             from 'firebase/auth';
import { getFirestore }        from 'firebase/firestore';
import { getReactNativePersistence, initializeAuth } from 'firebase/auth';
import ReactNativeAsyncStorage  from '@react-native-async-storage/async-storage';

const firebaseConfig = {
  apiKey:            "AIzaSy...",  // ← depuis console.firebase.google.com
  authDomain:        "monapp.firebaseapp.com",
  projectId:         "monapp-12345",
  storageBucket:     "monapp-12345.appspot.com",
  messagingSenderId: "123456789",
  appId:             "1:123456789:web:abc123",
};

const app = initializeApp(firebaseConfig);

export const auth = initializeAuth(app, {
  persistence: getReactNativePersistence(ReactNativeAsyncStorage)

🔒Ne jamais commiter firebaseConfig dans un dépôt public. En production, utiliser des variables d'environnement via .env et expo-constants.
🔐 Firebase Auth — email / password
💡 Analogie

Firebase Auth, c'est comme un portier automatique. Tu lui donnes un email + mot de passe, il vérifie dans sa liste et te donne un badge (token) si c'est bon. onAuthStateChanged est le portier qui surveille en permanence si le badge est toujours valide.

1

createUserWithEmailAndPassword(auth, email, password)

Créer un compte. Retourne une Promise avec UserCredential. L'utilisateur est automatiquement connecté après inscription.

2

signInWithEmailAndPassword(auth, email, password)

Se connecter avec un compte existant. Retourne une Promise avec UserCredential. En cas d'erreur, un code Firebase est disponible dans error.code.

3

signOut(auth)

Se déconnecter. Simple et synchrone. onAuthStateChanged sera notifié immédiatement avec null.

4

onAuthStateChanged(auth, callback)

Listener en temps réel qui écoute les changements d'état auth. Reçoit user (objet utilisateur) ou null (déconnecté). Remplace le booléen mock de J2.

App.js — onAuthStateChanged remplace le mock J2
import { useState, useEffect } from 'react';
import { onAuthStateChanged }  from 'firebase/auth';
import { auth }               from './services/firebase';

export default function App() {
  const [user,    setUser]    = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // ← Remplace le useState(false) mock de J2
    const unsub = onAuthStateChanged(auth, (currentUser) => {
      setUser(currentUser);
      setLoading(false);
    });
    return unsub; // cleanup — stop listening au démontage
  }, []);

  if (loading) return <LoadingScreen />;

  return (
    <NavigationContainer>
      {user ? <AppStack /> : <AuthStack />}
    </NavigationContainer>
  );
}

import { createUserWithEmailAndPassword } from 'firebase/auth';
import { doc, setDoc, serverTimestamp } from 'firebase/firestore';
import { auth, db } from '../services/firebase';

const handleRegister = async () => {
  try {
    // 1. Créer le compte Auth
    const { user } = await createUserWithEmailAndPassword(auth, email, password);

    // 2. Créer le profil dans Firestore
    await setDoc(doc(db, 'users', user.uid), {
      pseudo:    pseudo,
      email:     user.email,
      createdAt: serverTimestamp(),
    });
    // onAuthStateChanged() détecte la connexion → navigation automatique
  } catch (err) {
    setError(getFirebaseErrorMessage(err.code));
  }
};
import { signInWithEmailAndPassword } from 'firebase/auth';

const getFirebaseErrorMessage = (code) => {
  const messages = {
    'auth/invalid-email':       'Adresse email invalide',
    'auth/user-not-found':      'Aucun compte avec cet email',
    'auth/wrong-password':      'Mot de passe incorrect',
    'auth/email-already-in-use':'Cet email est déjà utilisé',
    'auth/weak-password':       'Mot de passe trop court (6 car. min)',
    'auth/too-many-requests':   'Trop de tentatives, réessaie plus tard',
  };
  return messages[code] || 'Erreur de connexion';
};

const handleLogin = async () => {
  try {
    await signInWithEmailAndPassword(auth, email, password);
    // onAuthStateChanged → navigation automatique vers AppStack
  } catch (err) {
    setError(getFirebaseErrorMessage(err.code));
  }
};

S6 · Firestore — CRUD & Temps réel

Modèle de données · getDocs · addDoc · deleteDoc · onSnapshot · Règles de sécurité

🏗 Modèle de données Firestore — Collections & Documents
💡 Analogie

Firestore ressemble à un classeur de dossiers. Une collection c'est un dossier (ex: "users"). Un document c'est une fiche dans ce dossier (ex: l'utilisateur "uid123"). Chaque fiche peut avoir des sous-dossiers (sous-collections). Pas de jointures SQL — on dénormalise les données.

📋 Opérations CRUD Firestore
R

Lire — getDocs(query) · getDoc(docRef)

Lecture ponctuelle (une seule fois). getDocs pour une liste, getDoc pour un document précis. Utiliser where() pour filtrer.

C

Créer — addDoc(collection, data) · setDoc(docRef, data)

addDoc génère un ID automatique. setDoc crée ou écrase un document avec un ID précis (ex: users/{'{uid}'}).

U

Modifier — updateDoc(docRef, fields)

Met à jour seulement les champs spécifiés — les autres restent intacts. Contrairement à setDoc qui écrase tout.

D

Supprimer — deleteDoc(docRef)

Supprime un document. Les sous-collections ne sont pas supprimées automatiquement — il faut les supprimer séparément si nécessaire.

CRUD — patterns essentiels
import {
  collection, doc, query, where, orderBy, limit,
  getDocs, getDoc, addDoc, setDoc, updateDoc, deleteDoc,
  onSnapshot, serverTimestamp
} from 'firebase/firestore';
import { db } from '../services/firebase';

// READ — lire les questions d'un thème
const q = query(
  collection(db, 'questions'),
  where('themeId', '==', 'T1'),
  limit(10)
);
const snapshot = await getDocs(q);
const questions = snapshot.docs.map(d => ({ id: d.id, ...d.data() }));

// CREATE — sauvegarder un score (ID auto-généré)
await addDoc(
  collection(db, 'users', user.uid, 'history'),
  { themeId, score, total, date: serverTimestamp() }
);

// UPDATE — mettre à jour le pseudo
await updateDoc(doc(db, 'users', user.uid), { pseudo: newPseudo });

// DELETE — supprimer un document
await deleteDoc(doc(db, 'items', itemId));

Exemples — cliquer pour développer :

Contrairement à getDocs (lecture unique), onSnapshot maintient une connexion active — l'UI se met à jour dès qu'un document change dans Firestore.

useEffect(() => {
  const q = query(
    collection(db, 'users', user.uid, 'history'),
    orderBy('date', 'desc'),
    limit(10)
  );

  // onSnapshot retourne une fonction de cleanup
  const unsub = onSnapshot(q, (snapshot) => {
    const history = snapshot.docs.map(d => ({ id: d.id, ...d.data() }));
    setHistory(history);  // ← mise à jour auto à chaque changement
  });

  return unsub; // cleanup — fermer la connexion au démontage
}, [user.uid]);
💡Toujours retourner la fonction de cleanup de onSnapshot dans le return du useEffect pour éviter les fuites mémoire.

🔒 Règles de sécurité Firestore
⚠️Le mode test (règles ouvertes) est dangereux en production. Il faut définir des règles précises avant le déploiement.
firestore.rules — règles ciblées App Quiz
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // themes et questions : lecture publique, écriture admin seulement
    match /themes/{themeId} {
      allow read:  if request.auth != null;
      allow write: if false;  // ← admin only (via console)
    }
    match /questions/{questionId} {
      allow read:  if request.auth != null;
      allow write: if false;
    }

    // users : chaque utilisateur accède uniquement à ses propres données
    match /users/{userId} {
      allow read, write: if request.auth != null
                         && request.auth.uid == userId;

      // sous-collection history : même règle
      match /history/{histId} {
        allow read, write: if request.auth != null
                           && request.auth.uid == userId;
      }
    }
  }
}

10 QCM — Vérification des acquis J3

Firebase Auth · Firestore · Règles de sécurité

📝Sélectionne une réponse par question puis clique sur "Valider mes réponses".
Question 1 / 10 · Firebase
Quelle est la différence entre Firestore et Realtime Database dans Firebase ?
Firestore est le service recommandé pour les nouvelles apps. Il supporte les requêtes complexes, l'indexation automatique, et scale mieux. Realtime DB est une arborescence JSON plus simple, utile pour des synchronisations ultra-rapides mais limitée en query.
Question 2 / 10 · Auth
Que retourne onAuthStateChanged(auth, callback) et pourquoi est-ce important ?
onAuthStateChanged() retourne une fonction unsubscribe. On DOIT la retourner dans le return du useEffect pour arrêter le listener quand le composant se démonte — sinon c'est une fuite mémoire. La callback reçoit user (objet) ou null (déconnecté).
Question 3 / 10 · Auth
Quelle est la différence entre setDoc() et addDoc() dans Firestore ?
setDoc(doc(db, 'users', uid), data) crée ou écrase le document à l'ID spécifié — idéal pour les profils utilisateurs (on veut l'ID = uid). addDoc(collection(db, 'history'), data) génère un ID aléatoire — idéal pour des entrées multiples comme l'historique.
Question 4 / 10 · Firestore
Quelle est la différence entre getDocs() et onSnapshot() ?
getDocs() fait une requête HTTP unique → retourne les données une fois. onSnapshot() ouvre une connexion WebSocket persistante → la callback est rappelée à chaque modification du document ou de la collection. Utiliser getDocs() pour charger les questions, onSnapshot() pour l'historique en temps réel.
Question 5 / 10 · Firestore
Comment structurer les données de l'historique de quiz dans Firestore ?
La sous-collection users/{uid}/history est le pattern recommandé. Elle isole les données par utilisateur (facilite les règles de sécurité), permet des requêtes ciblées (tri par date, limite), et scale bien (pas de tableau qui grandit indéfiniment dans un document).
Question 6 / 10 · Sécurité
En mode test Firestore, pourquoi les règles ouvertes sont-elles dangereuses en production ?
En mode test, les règles permettent tout. Le firebaseConfig est visible dans le code client (c'est normal — il n'est pas secret). Donc n'importe qui peut faire des requêtes directes à ton Firestore via le SDK ou l'API REST. Les règles ciblées sont la seule protection.
Question 7 / 10 · Sécurité
Que signifie request.auth.uid == userId dans les règles Firestore ?
request.auth.uid est l'uid de l'utilisateur qui fait la requête. userId est le segment de path (variable dans match /users/{userId}). L'égalité garantit qu'Alice ne peut pas lire/écrire les données de Bob. C'est la règle de base pour les données privées.
Question 8 / 10 · Auth
Pourquoi utiliser initializeAuth avec getReactNativePersistence(AsyncStorage) plutôt que getAuth simple ?
Sans persistance, la session Firebase est perdue à chaque fermeture d'app. Avec AsyncStorage, le token est sauvegardé localement. Au redémarrage, onAuthStateChanged() retrouve l'utilisateur automatiquement — pas besoin de se reconnecter à chaque lancement.
Question 9 / 10 · Firestore
Quelle est la bonne façon de charger les questions d'un thème spécifique depuis Firestore ?
Toujours filtrer côté serveur avec where() — c'est plus performant (moins de données transférées) et moins coûteux en quota Firestore. Filtrer côté client sur une collection entière est à éviter car ça télécharge tout avant de filtrer.
Question 10 / 10 · Architecture
Pourquoi ne pas stocker la clé apiKey Firebase dans un fichier .env et la mettre dans .gitignore ?
L'apiKey Firebase côté client n'est pas secrète — elle est visible dans le code compilé et c'est attendu. La vraie sécurité = les règles Firestore. En revanche, les clés d'administration Firebase (firebase-admin SDK, server-side) sont sensibles et ne doivent JAMAIS être dans le code client. Utiliser .env pour les garder hors du dépôt est une bonne pratique.

Glossaire — Termes clés J3

Firebase · Auth · Firestore · Sécurité

🔐 Firebase Auth

Firebase Auth Auth
Service d'authentification Firebase. Gère les comptes utilisateurs, les sessions, les tokens JWT. Prend en charge email/password, Google, Apple, etc.
Pas besoin de créer un serveur d'auth — Firebase gère tout.
onAuthStateChanged Auth
Listener Firebase qui écoute les changements d'état d'auth en temps réel. Retourne user (objet) ou null. Remplace le booléen mock de J2.
Retourner le cleanup dans useEffect pour éviter les fuites mémoire.
UserCredential Auth
Objet retourné par signIn() et createUser(). Contient .user (l'utilisateur) avec son uid, email, etc.
const {'{'} user {'}'} = await createUserWithEmailAndPassword(...)
uid Auth
Identifiant unique généré par Firebase pour chaque utilisateur. Utilisé comme clé de document dans Firestore (users/{'{uid}'}) et dans les règles de sécurité.
L'uid ne change jamais — même si l'utilisateur change d'email.
Persistance de session Auth
Mécanisme qui conserve la session Auth après fermeture de l'app. En React Native, on utilise AsyncStorage via getReactNativePersistence.
Sans persistance, l'utilisateur doit se reconnecter à chaque lancement.

🗄 Firestore

Collection Firestore
Conteneur de documents dans Firestore. Équivalent d'une table SQL, mais sans schéma fixe. Les collections peuvent être imbriquées (sous-collections).
collection(db, 'users')
Document Firestore
Unité de données dans Firestore — équivalent d'une ligne SQL. Contient des paires clé/valeur. Limité à 1 Mo. Peut contenir des sous-collections.
doc(db, 'users', uid)
onSnapshot Firestore
Listener temps réel Firestore. La callback est appelée immédiatement avec les données actuelles, puis à chaque modification. Retourne une fonction de cleanup.
Utiliser pour l'historique des scores en temps réel.
serverTimestamp() Firestore
Valeur spéciale qui est remplacée par l'horodatage du serveur Firebase lors de l'écriture. Plus fiable que new Date() côté client.
{'{'} createdAt: serverTimestamp() {'}'}
query() + where() Firestore
Filtrage côté serveur. query(collection, where('champ', '==', valeur)) ne télécharge que les documents correspondants — bien plus efficace que filtrer côté client.
where('themeId', '==', 'T1')
Règles de sécurité Firestore
Langage déclaratif qui contrôle les accès à Firestore côté serveur. Évaluées par Firebase — impossible à contourner côté client.
allow read, write: if request.auth.uid == userId;

🏗 Architecture

Seeding Architecture
Opération d'initialisation de la base de données avec des données de départ. En Firestore, se fait via un script ou la console Firebase.
Seeder les questions et thèmes avant les tests de l'app Quiz.
Dénormalisation Firestore
Pratique Firestore (et NoSQL en général) qui consiste à dupliquer des données pour éviter les jointures. Ex: stocker le nom du thème dans chaque entrée d'historique.
Contrairement au SQL où on évite la duplication.
Mode test vs Production Sécurité
Mode test = règles ouvertes (tout le monde peut tout faire). Mode production = règles ciblées basées sur l'auth. Toujours passer en mode production avant le déploiement.
Le mode test expire automatiquement après 30 jours.
